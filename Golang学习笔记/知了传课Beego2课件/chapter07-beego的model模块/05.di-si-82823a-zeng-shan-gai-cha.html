演示代码文件：controllers/test_orm/test_orm_chap01.go

新增一条：Insert
o := orm.NewOrm()
o.Using("default")

newStudent := Student{Name:"zhiliao",Age:18,Gender:"男"}

//新增数据
id, err :=o.Insert(&newStudent)    // 返回添加的数据id以及错误信息
新增多条：InsertMulti
o.InsertMulti(bulk, slice对象)        第一个参数 bulk 为并列插入的数量，第二个为对象的slice


o := orm.NewOrm()
o.Using("default")

newStudent := []Student{{Name:"zhiliao2",Age:20,Gender:"男"},{Name:"zhiliao3",Age:19,Gender:"男"}}

num, _ :=o.InsertMulti(100,newStudent)      // 返回的插入的条数和错误
fmt.Println(num)
查询一条数据：Read
o := orm.NewOrm()
o.Using("default")

newStudent := Student{Id:4}   // 查询id=4的数据

err := o.Read(&newStudent)
if err == orm.ErrNoRows {     // 查看orm.go中定义的错误
  fmt.Println("查询不到")
}
// 打印查询到的信息
fmt.Println(newStudent)



注意：
默认是根据主键查询的，如果要根据其他字段查询，需要指定字段名

指定一个字段：
newStudent := Student{Name:"zhiliao2"} // 
err := o.Read(&newStudent,"Name")

指定多个字段：
newStudent := Student{Id:14,Name:"zhiliao2"} // 查询id=4的数据
err := o.Read(&newStudent,"Id","Name")    // clos是个不定长参数
查询不到则创建：ReadOrCreate
newStudent := Student{Id:16,Name:"hailong",Age:18,Gender:"男"} // 查询id=16的数据

is_new,id,err := o.ReadOrCreate(&newStudent,"Id")       // 注意这里必须指定查询的字段
fmt.Println(is_new)
fmt.Println(id)
fmt.Println(err)

返回三个参数：是否新创建的(bool)，Id的值，错误信息
更新：Update
先查询再更新
newStudent := Student{Name:"hallen"}
err := o.Read(&newStudent,"Name")
if err == nil{
    newStudent.Name = "hailong"
    newStudent.Gender = "女"
    num ,err := o.Update(&newStudent)    // 第一个返回的是受影响的行数，第二个是错误
    fmt.Println(num)
    fmt.Println(err)

}

对指定的字段更新，默认是全部更新：可以指定多个，不定长参数

num ,err := o.Update(&newStudent,"Name","Gender")
删除：Delete
默认是根据id查询删除
newStudent := Student{Id:2}
num, err := o.Delete(&newStudent)
fmt.Println(num)
fmt.Println(err)

可以指定列名：
newStudent := Student{Name:"hailong"}
num, err := o.Delete(&newStudent,"Name")
fmt.Println(num)
fmt.Println(err)
添加或更新：InsertOrUpdate 没有则添加，有则更新

注意：

一定要在模型中指定唯一的字段，然后在使用的时候指定这个唯一字段
